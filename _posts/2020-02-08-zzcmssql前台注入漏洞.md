```php
if($_REQUEST){

			//@extract($_GET);

			$_POST = zc_check($_POST);
			$_GET = zc_check($_GET);
			$_COOKIE =zc_check($_COOKIE);
			//@extract($_POST);
			//@extract($_GET);
		}
}
```
